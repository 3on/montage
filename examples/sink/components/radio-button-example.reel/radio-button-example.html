RadioButton

The RadioButton component wraps an HTML radio utton input type with custom styles and functionality.

Binding example

What is your favorite food?

Serialization
"veg": {
    "module": "montage/ui/radio-button.reel",
    "name": "RadioButton",
    "properties": {
        "element": {"#": "veg"}
    }
},

"veg-binding": {
    "value": {
        "disabled": {
            "boundObject": {"@": "veg"},
            "boundObjectPropertyPath": "checked",
            "oneway": true,
            "converter": {"@": "invert-converter"}
        }
    }
},

"veg-carrot": {
    "module": "montage/ui/radio-button.reel",
    "name": "RadioButton",
    "properties": {
        "element": {"#": "veg-carrot"}
    },
    "bindings": {"@": "veg-binding"}
}
// Repeated for each vegetable and fruit
